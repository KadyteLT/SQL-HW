SELECT `news`.`newsId`, `news`.`text`, `news`.`date`, `topComment`.`commentId`, `topComment`.`text` AS commentText, `topComment`.`date` AS commentDate FROM `news` LEFT JOIN ( SELECT * FROM ( SELECT * FROM `comments` ORDER BY `date` DESC LIMIT 18446744073709551615 ) AS `ordered` GROUP BY `ordered`.`newsId` ) AS `topComment` ON `topComment`.`newsId` = `news`.`newsId` ORDER BY `news`.`date` DESC;
